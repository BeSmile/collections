# 图片上传
监听onChange事件，利用FileReader获取base64
<pre>
    _handleImageChange(e) {
        e.preventDefault();

        let reader = new FileReader();
        let file = e.target.files[0];

        reader.onloadend = () => {
            this.setState({

                imagePreviewUrl: reader.result
            });

        };
        reader.readAsDataURL(file)
    }
</pre>
